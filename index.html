// Function to send a message async function sendMessage() { const message = document.getElementById('messageInput').value; if (message.trim() === '') return; // Do nothing if the message is empty await db.collection('messages').add({ text: message, timestamp: firebase.firestore.FieldValue.serverTimestamp() }); document.getElementById('messageInput').value = ''; // Clear the input field } // Function to display messages db.collection('messages').orderBy('timestamp').onSnapshot(snapshot => { const messagesDiv = document.getElementById('messages'); messagesDiv.innerHTML = ''; // Clear the messages div snapshot.forEach(doc => { const message = doc.data().text; messagesDiv.innerHTML += `

${message}

`; }); messagesDiv.scrollTop = messagesDiv.scrollHeight; // Scroll to the bottom });
